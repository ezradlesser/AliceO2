45 void ITSDCSParser::run(ProcessingContext& pc)
46 {
47  // check O2/Detectors/DCS/testWorkflow/src/dcs-config-proxy.cxx for input format
48 
49  // Retrieve string from inputs
50  const auto inString = pc.inputs().get<gsl::span<char>>("inString");
51  std::string inStringConv;
52  std::copy(inString.begin(), inString.end(), std::back_inserter(inStringConv));
53  for (const std::string& line : this->vectorizeStringList(inStringConv, "\n")) {
54  if (!line.length()) {
55  continue;
56  }
57  this->updateMemoryFromInputString(line);
58  this->saveToOutput();
59  this->resetMemory();
60  }
61 
62  if (this->mConfigDCS.size()) {
63  this->pushToCCDB(pc);
64  this->mConfigDCS.clear();
65  }
66 
67  return;
68 }

71 void ITSDCSParser::updateMemoryFromInputString(const std::string& inString)
72 {
73  // Parse the individual parts of the string
74  const std::string delimiter = "|";
75  unsigned int pos = 0;
76  unsigned int npos = inString.find(delimiter);
77 
78  // First is the stave name
79  this->mStaveName = inString.substr(pos, npos);
80 
81  // Next is the run number
82  std::string word = "RUN";
83  pos += npos + delimiter.length() + word.length();
84  npos = inString.find(delimiter, pos) - pos;
85  this->updateAndCheck(this->mRunNumber, std::stoi(inString.substr(pos, npos)));
86 
87  // Next is the config
88  word = "CONFIG";
89  pos += npos + delimiter.length() + word.length();
90  npos = inString.find(delimiter, pos) - pos;
91  this->updateAndCheck(this->mConfigVersion, std::stoi(inString.substr(pos, npos)));
92 
93  // Then it's the run type
94  word = "RUNTYPE";
95  pos += npos + delimiter.length() + word.length();
96  npos = inString.find(delimiter, pos) - pos;
97  this->updateAndCheck(this->mRunType, std::stoi(inString.substr(pos, npos)));
98 
99  // Then there is a semicolon-delineated list of disabled chips
100  word = "DISABLED_CHIPS";
101  pos += npos + delimiter.length() + word.length();
102  npos = inString.find(delimiter, pos) - pos;
103  std::string disabledChipsStr = inString.substr(pos, npos);
104  if (disabledChipsStr.length()) {
105  this->mDisabledChips = this->vectorizeStringListInt(disabledChipsStr, ";");
106  }
107 
108  // Then there is a 2D list of masked double-columns
109  word = "MASKED_DCOLS";
110  pos += npos + delimiter.length() + word.length();
111  npos = inString.find(delimiter, pos) - pos;
112  std::string maskedDoubleColsStr = inString.substr(pos, npos);
113  if (maskedDoubleColsStr.length()) {
114  std::vector<std::string> chipVect = this->vectorizeStringList(maskedDoubleColsStr, ";");
115  for (const std::string& str : chipVect) {
116  // Element 0 in each subvector is chip ID, rest are double column numbers
117  this->mMaskedDoubleCols.push_back(this->vectorizeStringListInt(str, ":"));
118  }
119  }
120 
121  // Finally, there are double columns which are disabled at EOR
122  word = "DCOLS_EOR";
123  pos += npos + delimiter.length() + word.length();
124  std::string doubleColsEORstr = inString.substr(pos);
125  if (doubleColsEORstr.length()) {
126  std::vector<std::string> bigVect = this->vectorizeStringList(doubleColsEORstr, "&");
127  for (const std::string& bigStr : bigVect) {
128  std::vector<std::string> bigVectSplit = this->vectorizeStringList(bigStr, "|");
129  if (!bigVectSplit.size()) {
130  continue;
131  }
132 
133  // First, update map of disabled double columns at EOR
134  std::vector<std::string> doubleColDisable = this->vectorizeStringList(bigVectSplit[0], ";");
135  for (const std::string& str : doubleColDisable) {
136  std::vector<unsigned short int> doubleColDisableVector = this->vectorizeStringListInt(str, ":");
137  this->mDoubleColsDisableEOR[doubleColDisableVector[0]].push_back(doubleColDisableVector[1]);
138  }
139  // Second, update map of flagged pixels at EOR
140  std::vector<std::string> pixelFlagsEOR = this->vectorizeStringList(bigVectSplit[1], ";");
141  for (const std::string& str : pixelFlagsEOR) {
142  std::vector<unsigned short int> pixelFlagsVector = this->vectorizeStringListInt(str, ":");
143  this->mPixelFlagsEOR[pixelFlagsVector[0]].push_back(pixelFlagsVector[1]);
144  }
145  }
146  }
147 
148  return;
149 }

152 // Update memValue using newValue if memValue is not yet set
153 void ITSDCSParser::updateAndCheck(int& memValue, const int newValue)
154 {
155  // Check if value in memory is nonsense, meaning it has never been updated
156  if (memValue == UNSET_INT) {
157  // Save value in memory for the first time (should always be the same)
158  memValue = newValue;
159  } else if (memValue != newValue) {
160  // Different value received than the one saved in memory -- throw error
161  throw newValue;
162  }
163 
164  return;
165 }

183 // Take string delimited by delimiter and parse into vector of objects
184 std::vector<std::string> ITSDCSParser::vectorizeStringList(
185  const std::string& str, const std::string& delimiter)
186 {
187  std::vector<std::string> str_vect;
188  std::string::size_type prev_pos = 0, pos = 0;
189  while ((pos = str.find(delimiter, pos)) != std::string::npos) {
190  std::string substr = str.substr(prev_pos, pos - prev_pos);
191  if (substr.length()) {
192  str_vect.push_back(substr);
193  }
194  pos += delimiter.length();
195  prev_pos = pos;
196  }
197  std::string substr = str.substr(prev_pos, pos - prev_pos);
198  if (substr.length()) {
199  str_vect.push_back(substr);
200  }
201  return str_vect;
202 }

205 std::vector<unsigned short int> ITSDCSParser::vectorizeStringListInt(
206  const std::string& str, const std::string& delimiter)
207 {
208  std::vector<unsigned short int> int_vect;
209  std::string::size_type prev_pos = 0, pos = 0;
210  while ((pos = str.find(delimiter, pos)) != std::string::npos) {
211  unsigned short int substr = std::stoi(str.substr(prev_pos, pos - prev_pos));
212  int_vect.push_back(substr);
213  pos += delimiter.length();
214  prev_pos = pos;
215  }
216  int_vect.push_back(std::stoi(str.substr(prev_pos, pos - prev_pos)));
217  return int_vect;
218 }

221 void ITSDCSParser::saveToOutput()
222 {
223  // First loop through the disabled chips to write these to the string
224  for (const unsigned short int& chipID : this->mDisabledChips) {
225  // Write basic chip info
226  this->writeChipInfo(this->mConfigDCS, this->mStaveName, chipID);
227 
228  // Mark chip as disabled
229  o2::dcs::addConfigItem(this->mConfigDCS, "Disabled", "1");
230 
231  // Mark other information with nonsense value
232  o2::dcs::addConfigItem(this->mConfigDCS, "Dcol_masked", "-1");
233  o2::dcs::addConfigItem(this->mConfigDCS, "Dcol_masked_eor", "-1");
234  o2::dcs::addConfigItem(this->mConfigDCS, "Pixel_flags", "-1");
235 
236  // Ensure that chips are removed from the maps
237  mDoubleColsDisableEOR.erase(chipID);
238  mPixelFlagsEOR.erase(chipID);
239  }
240 
241  // Second, loop through all the chips with disabled double columns
242  for (std::vector<unsigned short int> maskedDoubleCols : this->mMaskedDoubleCols) {
243  unsigned short int chipID = maskedDoubleCols[0];
244  maskedDoubleCols.erase(maskedDoubleCols.begin());
245 
246  // Write basic chip info
247  this->writeChipInfo(this->mConfigDCS, this->mStaveName, chipID);
248  o2::dcs::addConfigItem(this->mConfigDCS, "Disabled", "0");
249 
250  // Write information for disabled double columns
251  o2::dcs::addConfigItem(this->mConfigDCS, "Dcol_masked", this->intVecToStr(maskedDoubleCols, "|"));
252 
253  // Retrieve information from maps, if any, and then erase
254  o2::dcs::addConfigItem(
255  this->mConfigDCS, "Dcol_masked_eor", this->intVecToStr(this->mDoubleColsDisableEOR[chipID], "|"));
256  this->mDoubleColsDisableEOR.erase(chipID);
257  o2::dcs::addConfigItem(
258  this->mConfigDCS, "Pixel_flags", this->intVecToStr(this->mPixelFlagsEOR[chipID], "|"));
259  this->mPixelFlagsEOR.erase(chipID);
260  }
261 
262  // Finally, loop through any remaining chips
263  for (const auto& [chipID, v] : this->mDoubleColsDisableEOR) {
264  std::string s = this->intVecToStr(v, "|");
265  if (s != "-1") { // Ensure no meaningless entries
266  this->writeChipInfo(this->mConfigDCS, this->mStaveName, chipID);
267  o2::dcs::addConfigItem(this->mConfigDCS, "Disabled", "0");
268  o2::dcs::addConfigItem(this->mConfigDCS, "Dcol_masked", "-1");
269  o2::dcs::addConfigItem(this->mConfigDCS, "Dcol_masked_eor", this->intVecToStr(v, "|"));
270  o2::dcs::addConfigItem(
271  this->mConfigDCS, "Pixel_flags", this->intVecToStr(this->mPixelFlagsEOR[chipID], "|"));
272  this->mPixelFlagsEOR.erase(chipID);
273  }
274  }
275 
276  for (const auto& [chipID, v] : this->mPixelFlagsEOR) {
277  std::string s = this->intVecToStr(v, "|");
278  if (s != "-1") { // Ensure no meaningless entries
279  this->writeChipInfo(this->mConfigDCS, this->mStaveName, chipID);
280  o2::dcs::addConfigItem(this->mConfigDCS, "Disabled", "0");
281  o2::dcs::addConfigItem(this->mConfigDCS, "Dcol_masked", "-1");
282  o2::dcs::addConfigItem(this->mConfigDCS, "Dcol_masked_eor", "-1");
283  o2::dcs::addConfigItem(this->mConfigDCS, "Pixel_flags", this->intVecToStr(v, "|"));
284  }
285  }
286 
287  return;
288 }

303 // Convert vector of integers to one single string, delineated by delin
304 std::string ITSDCSParser::intVecToStr(
305  const std::vector<unsigned short int>& v, const std::string& delin)
306 {
307  // Return nonsense value if there are no entries
308  if (v.empty()) {
309  return "-1";
310  }
311 
312  std::string bigStr = std::to_string(v[0]);
313  for (int i = 1; i < v.size(); i++) {
314  bigStr += delin + std::to_string(v[i]);
315  }
316 
317  return bigStr;

321 // Write the Stave, Hs_pos, Hic_pos, and ChipID to the config object
322 void ITSDCSParser::writeChipInfo(
323  o2::dcs::DCSconfigObject_t& configObj, const std::string& staveName,
324  const unsigned short int chipID)
325 {
326  // First save the Stave to the string
327  o2::dcs::addConfigItem(configObj, "Stave", staveName);
328 
329  // Hs_pos: 0 is lower, 1 is upper
330  // Hic_pos: from 1 to 7 (module number)
331  unsigned short int hicPos = chipID / 16;
332  bool hsPos = (hicPos > 7);
333  if (hsPos) {
334  hicPos -= 8;
335  }
336  o2::dcs::addConfigItem(configObj, "Hs_pos", std::to_string(hsPos));
337  o2::dcs::addConfigItem(configObj, "Hic_pos", std::to_string(hicPos));
338 
339  // Chip ID inside the module
340  o2::dcs::addConfigItem(configObj, "ChipID", std::to_string(chipID % 16));
341 
342  return;
343 }

346 void ITSDCSParser::pushToCCDB(ProcessingContext& pc)
347 {
348  // Timestamps for CCDB entry
349  long tstart = o2::ccdb::getCurrentTimestamp();
350  long tend = tstart + 365L * 24 * 3600 * 1000;
351 
352  auto class_name = o2::utils::MemFileHelper::getClassName(mConfigDCS);
353 
354  // Create metadata for database object
355  std::map<std::string, std::string> md = {
356  {"runtype", std::to_string(this->mRunType)}, {"confDBversion", std::to_string(this->mConfigVersion)}};
357  if (!mCcdbUrl.empty()) { // add only if we write here otherwise ccdb-populator-wf add it already
358  md.insert({"runNumber", std::to_string(this->mRunNumber)});
359  }
360  std::string path("ITS/DCS_CONFIG/");
361  const char* filename = "dcs_config.root";
362  o2::ccdb::CcdbObjectInfo info(path, "dcs_config", filename, md, tstart, tend);
363  auto image = o2::ccdb::CcdbApi::createObjectImage(&mConfigDCS, &info);
364  info.setFileName(filename);
365 
366  // Send to ccdb-populator wf
367  LOG(info) << "Class Name: " << class_name << " | File Name: " << filename
368  << "\nSending to ccdb-populator the object " << info.getPath() << "/" << info.getFileName()
369  << " of size " << image->size() << " bytes, valid for "
370  << info.getStartValidityTimestamp() << " : "
371  << info.getEndValidityTimestamp();
372 
373  pc.outputs().snapshot(Output{o2::calibration::Utils::gDataOriginCDBPayload, "DCS_CONFIG", 0}, *image);
374  pc.outputs().snapshot(Output{o2::calibration::Utils::gDataOriginCDBWrapper, "DCS_CONFIG", 0}, info);
375 
376  if (!mCcdbUrl.empty()) { // if url is specified, send object to ccdb from THIS wf
377 
378  LOG(info) << "Sending object " << info.getFileName() << " to " << mCcdbUrl << "/browse/"
379  << info.getPath() << " from the ITS string parser workflow";
380  o2::ccdb::CcdbApi mApi;
381  mApi.init(mCcdbUrl);
382  mApi.storeAsBinaryFile(
383  &image->at(0), image->size(), info.getFileName(), info.getObjectType(), info.getPath(),
384  info.getMetaData(), info.getStartValidityTimestamp(), info.getEndValidityTimestamp());
385  }
386 
387  return;
388 }
